f_id = []

r_slot = []

for i in passes.index:
    for form in forms.index:
        if i >= form: # both DFs share the same index values, so can use the starting index of each formation/tactic to know when to go to the next tactic
            tactic = forms.id[form]
            f_name = int(forms.tactics_formation[form])
            
            a = players[players.player_id == passes.player_id[i]].copy(deep=True)
            if len(a[a.id == tactic]) != 0: # make sure player exists in tactic
                p = a[a.id == tactic]['player_position_id'].values[0] # store slot id for this player in this tactic

                b = players[players.player_id == passes.pass_recipient_id[i]].copy(deep=True)
                if len(b) == 0: # check player receiving pass exists in tactic. will return a blank df if no recipient
                    r = 'No Recipient'
                elif len(b[b.id == tactic]) != 0: # confirm tactic from passes df exists in players df
                    r = b[b.id == tactic]['player_position_id'].values[0] # store slot id for this player in this tactic
        else:
            continue
    f_id.append(tactic)
    f_names.append(f_name)
    p_slot.append(p)
    r_slot.append(r)

print([len(x) for x in [passes.index, f_id, f_names, p_slot, r_slot]]) # check these are all the same

slot_actions = pd.DataFrame()

for i in form_dict.keys():
    mask1 = player_locs.player_slot == i
    mask2 = player_locs.f_name == formation
    avg_x = player_locs[mask1 & mask2]['x'].mean()
    avg_y = player_locs[mask1 & mask2]['y'].mean()
    count = len(player_locs[mask1 & mask2])
    df = pd.DataFrame(data = {'slot':[i], 'avg_x':[avg_x], 'avg_y':[avg_y], 'event_count':[count]})
    slot_actions = slot_actions.append(df, ignore_index=True)

Create DF For Passes Between Each Slot¶

In [13]:
slot_passes = pd.DataFrame()

for passer in form_dict.keys():
    mask1 = passes.player_slot == passer
    mask2 = passes.outcome_name.isna()
    mask3 = passes.f_name == formation
    for recipient in slot_actions.slot.unique():
        mask4 = passes.player_slot == recipient
        mask5 = passes.recipient_slot == recipient
        mask6 = passes.recipient_slot == passer
        df = passes[mask2 & mask3 & ((mask1 & mask5) | (mask4 & mask6))] # add passes in either direction for pairing
        count = len(df)
        if count != 0: # we want to ignore rows where there were no passes between the players
            if len(slot_passes) > 0: # check against previously appended rows in slot_passes to look for opposite passer/recipient combo
                slot_pair_check = [] # we will append the boolean result to this list
                for i in slot_passes.index:
                    if slot_passes.passer_slot[i] == recipient and slot_passes.recipient_slot[i] == passer:
                        slot_pair_check.append(True)
                    else:
                        slot_pair_check.append(False)
                if True in slot_pair_check:
                    continue
                df2 = pd.DataFrame(data = {'passer_slot':[passer], 'recipient_slot':[recipient], 'pass_count':[count]})
                slot_passes = slot_passes.append(df2, ignore_index = True)
            else: # didn't need to check the first combination for it's duplicate, but still need to add to df
                df2 = pd.DataFrame(data = {'passer_slot':[passer], 'recipient_slot':[recipient], 'pass_count':[count]})
                slot_passes = slot_passes.append(df2, ignore_index = True)

Plot the Pitch, Avg Slot Positions, and Passes¶

pitch = Pitch(pitch_type='statsbomb', orientation='horizontal',
              pitch_color='#22312b', line_color='#c7d5cc', figsize=(16,11),
              constrained_layout=True, tight_layout=False)
fig, ax = pitch.draw()

# plot pass lines between slot pairings, with the width relative to the pass_count
for i in slot_passes.index:
    p = slot_passes.passer_slot[i]
    r = slot_passes.recipient_slot[i]
    p_x = slot_actions[slot_actions.slot == p]['avg_x']
    p_y = slot_actions[slot_actions.slot == p]['avg_y']
    r_x = slot_actions[slot_actions.slot == r]['avg_x']
    r_y = slot_actions[slot_actions.slot == r]['avg_y']
    width = slot_passes.pass_count[i] * 0.5 # use a modifier (like 0.5) to keep the lines under control!
    pitch.lines(p_x, p_y, r_x, r_y, color='white', lw = width, ax = ax, zorder=1)

 # plot position marker for each slot, with size relative to number of actions in the match   
for i in slot_actions.index:
    if slot_actions['slot'][i] != 0:
        pos = form_dict[slot_actions.slot[i]]
        size = slot_actions.event_count[i] * 8 # again, use a modifier to control marker size
        pitch.scatter(slot_actions.avg_x[i], slot_actions.avg_y[i], color='red', edgecolors='black', linewidth=1, alpha=1, s=size, ax=ax)
        pitch.annotate(pos, xy=(slot_actions.avg_x[i], slot_actions.avg_y[i]),
                       xytext=(slot_actions.avg_x[i], slot_actions.avg_y[i]),
                       c='white', va='center', ha='center',size=16, weight='bold', ax=ax)
        
ax.set_title("{} {} Shape vs {}".format(team, formation, opponent),size=28)
